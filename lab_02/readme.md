## Программа
```asm
stackSEG SEGMENT PARA STACK 'STACK' ; Определяет сегмент стека с именем stackSEG. Атрибут PARA указывает
    ; что сегмент должен быть выровнен по границе параграфа (16 байт). STACK 'STACK' указывает, что это сегмент стека.
    DB 200h DUP (?) ; Резервирует 512 байт (200h в шестнадцатеричной системе) для стека. DUP (?) означает, что память не инициализирована.
    StkSeg ENDS ; Завершает определение сегмента стека.
stackSEG ENDS

dataSEG SEGMENT WORD 'DATA' ; Определяет сегмент данных с именем DataS. Атрибут WORD указывает, что данные выровнены по границе слова (2 байта).
HelloMessage DB 13 ; Возрат каретки
    DB 10 ; Перенос строки
    DB 'Hello World!' ; Строка
    DB '$' ; Ограничитель строки
dataSEG ENDS

codeSEG SEGMENT WORD 'CODE' ; Определяет сегмент кода с именем Code. Атрибут WORD указывает, что код выровнен по границе слова.
    ASSUME CS:codeSEG, DS:dataSEG ;  Указывает ассемблеру, что регистр CS будет указывать на сегмент codeSEG, а DS — на сегмент dataSEG.
dispMsg:
    mov ax, dataSEG ;загрузка в AX адреса сегмента данных
    mov DS, ax ; Загрузка в DS

    mov dx, offset HelloMessage ; Загружает в регистр DX смещение (адрес) строки HelloMessage в сегменте данных.
    mov ah, 9 ; Помещаем в регистр AH команду 9 (Вывод на экран)
    int 21h ; Системное прерывание которое выполняет функцию DOS, указанную в AH
    int 21h
    int 21h

    mov ah, 4ch ; Помещаем в регистр команду на завершение программы
    int 21h ; Системное прерывание которое выполняет функцию DOS, указанную в AH
codeSEG ENDS ; Завершает определение сегмента кода.
END dispMsg ; Указывает ассемблеру, что выполнение программы начинается с метки
```
